// SPEECH PROCESSING ALGORITHM LINKED TO THE OFFICIAL LIS CACHE function processVerbalOfficerCommand(spokenPhrase) { const promptLabel = document.getElementById('voicePromptText'); const textLower = spokenPhrase.toLowerCase(); let searchTargetCode = ""; let responseAudioPhrase = ""; if (textLower.includes("assault") || textLower.includes("57") || textLower.includes("battery")) { searchTargetCode = "§ 18.2-57"; responseAudioPhrase = "Virginia Code Section 18.2-57. Assault and Battery Against Law Enforcement. State LIS database indicates that if any individual commits assault against an officer while in the performance of their duty, it is a Class 6 felony."; } else if (textLower.includes("brandish") || textLower.includes("weapon") || textLower.includes("282")) { searchTargetCode = "§ 18.2-282"; responseAudioPhrase = "Virginia Code Section 18.2-282. Brandishing a Firearm. State law indicates it shall be unlawful to point, hold or brandish any firearm or object similar in appearance in such manner as to reasonably induce fear in the mind of another."; } else if (textLower.includes("reckless") || textLower.includes("traffic") || textLower.includes("852")) { searchTargetCode = "§ 46.2-852"; responseAudioPhrase = "Virginia Code Section 46.2-852. Reckless Driving. State traffic parameters dictate that any person who operates a motor vehicle on any highway in a manner that endangers life, limb, or property is guilty of reckless driving."; } else { searchTargetCode = "GENERAL QUERY"; responseAudioPhrase = "Command phrase registered via LIS tracking portal, but no direct statute code match was isolated. Please select a primary grid card category on screen."; } // Update the on-screen text box cleanly promptLabel.innerText = `📢 STREAMING LIS CODE: ${searchTargetCode}`; // Stop the microphone completely BEFORE starting the speakers so the phone doesn't mute itself if (voiceRecognitionInstance) { voiceRecognitionInstance.stop(); } // Fire the voice response speakStatuteAloud(responseAudioPhrase); } function speakStatuteAloud(phrase) { window.speechSynthesis.cancel(); // Force clear any jammed audio blocks const utterance = new SpeechSynthesisUtterance(phrase); utterance.rate = 0.95; // Maintain a clear, tactical speaking pace // Hide the console block ONLY when she is completely done talking utterance.onend = () => { document.getElementById('voiceConsole').style.display = 'none'; }; window.speechSynthesis.speak(utterance); }
